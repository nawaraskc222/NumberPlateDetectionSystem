import tkinter as tk

    label.config(text="Enter key pressed!")

root = tk.Tk()
root.title("Tkinter Keyboard Listener")

label = tk.Label(root, text="Press Enter key!")

//calculator


import tkinter as tk

def add_numbers():
    try:
        # Get the values from the entry widgets
        num1 = float(entry_num1.get())
        num2 = float(entry_num2.get())

        # Perform the addition
        result = num1 + num2

        # Display the result in the label
        result_label.config(text=f"Result: {result}", fg="green")
    except ValueError:
        # Handle the case where input is not a valid number
        result_label.config(text="Invalid input. Please enter numbers.", fg="red")

# Creating a main window
root = tk.Tk()
root.title("Simple Calculator")

# Setting a background color
root.configure(bg="#f0f0f0")

# Adding entry widgets for input
entry_num1 = tk.Entry(root, width=10, font=("Arial", 12))
entry_num1.grid(row=0, column=0, padx=10, pady=10)

entry_num2 = tk.Entry(root, width=10, font=("Arial", 12))
entry_num2.grid(row=0, column=1, padx=10, pady=10)

# Styling the button
add_button = tk.Button(root, text="Add", command=add_numbers, font=("Arial", 12), bg="#4CAF50", fg="white")
add_button.grid(row=1, column=0, columnspan=2, pady=10)

# Adding a label to display the result
result_label = tk.Label(root, text="Result: ", font=("Arial", 14), bg="#f0f0f0")
result_label.grid(row=2, column=0, columnspan=2, pady=10)

# Running the main loop
root.mainloop()
